# training-server
Частичный OpenSource training-server.com

Что в себе содержит pData[playerid]:

	Float: pPos[4], // Последние валидные XYZA координаты
	Float: pHealth, // Здоровье игрока
	Float: pArmour, // Броня игрока
	pInterior, // Интерьер игрока
	pSpectator, // За кем следит игрок
	bool: pInSpec, // В спекет игрок или нет
	bool: pTazer, // У игрока взят /taser
	pStunned, // Если игрок парализован, то на сколько секунд.
	bool: pLoopAnim, // У игрока зацикленная анимация
	pChannel, // Канал рации игрока
	pStyle, // Стиль боя игрока
	pActor, // Последний выделенный игроком актер
	pHit, // Количество попаданий выстрелов
	pShot, // Число выстрелов
	pWalkStyle, // Какой стиль походки у игрока
	bool: pDead, // Игрок метрв или нет
	pDeadStage, // Время стадии смерти игрока
	pID, // Глобальный ID игрока
	pFreezeTime, // Время заморозки игрока в МС после телепортов.
	bool: Frozen, // Заморожен игрок или нет
	pRegdate, // Дата регистрации игрока в timestamp формате
	vip, // ВИП игрок или нет
	aMute, // Есть у игрока мут или нет (от админов)
	bool: GM, // Включен ГМ у игрока или нет
	pAFK, // Время в АФК игрок
	pModer, // Лвл моедрки игрока на сервере
	pPickup, // ID последнего пикапа на котором стоял игрок.
	pSkills[12], // Скиллы игрока на каждый стиль оружия
	bool: pHideName, // Отображаются ники для игрока или нет
	pLastGun, // ID оружия с которого последний выстрел сделал игрок
	pVerify, // Ирок верефицирован или нет
	pChatStyle, // Стиль чата игрока (анимка общения)
	pTeam, // ID группы игрока
	bool: pBanned, // Забанен игрок в мини-играх или нет (активный Warn)
	pMoney, // Деньги игрока
	cbData[27], // Массив игрока
	pTakeDamageBPart, // В какую часть тела игрок получил урон
	pTakeDamageWeapon, // С какого оружия был нанесен игроку урон
	pRetVal[13], // Возвращаемые параметры для КБ
	pRetStr[264], // Возвращаемая строка для Кб
